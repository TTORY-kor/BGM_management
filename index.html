/** * BGM Library API for Google Sheets * * 시트 구조 * A: id * B: title * C: url * D: tag * E: folder * F: createdAt * G: updatedAt * * 첫 줄은 헤더 */ const SHEET_NAME = 'Sheet1'; // 실제 시트명으로 변경 function doGet(e) { const action = e && e.parameter ? String(e.parameter.action || '').trim() : ''; if (action === 'list') { return handleList(); } return jsonOutput({ ok: true, message: 'BGM Library API is running', timestamp: new Date().toISOString() }); } function doPost(e) { try { if (!e || !e.postData || !e.postData.contents) { return jsonOutput({ ok: false, error: '요청 본문이 없습니다.' }); } const data = JSON.parse(e.postData.contents || '{}'); const action = String(data.action || '').trim(); if (!action) { return jsonOutput({ ok: false, error: 'action 값이 없습니다.' }); } switch (action) { case 'ping': return jsonOutput({ ok: true, message: 'pong' }); case 'add': return handleAdd(data); case 'edit': return handleEdit(data); case 'delete': return handleDelete(data); default: return jsonOutput({ ok: false, error: `지원하지 않는 action 입니다: ${action}` }); } } catch (error) { return jsonOutput({ ok: false, error: String(error && error.message ? error.message : error) }); } } function handleList() { const sheet = getSheet_(); const lastRow = sheet.getLastRow(); const lastColumn = 7; if (lastRow < 2) { return jsonOutput({ ok: true, items: [] }); } const values = sheet.getRange(2, 1, lastRow - 1, lastColumn).getValues(); const items = values .map(row => ({ id: row[0], title: row[1], url: row[2], tag: row[3], folder: row[4], createdAt: row[5], updatedAt: row[6] })) .filter(item => item.id && item.url); return jsonOutput({ ok: true, items }); } function handleAdd(data) { const sheet = getSheet_(); const title = sanitizeText_(data.title); const url = sanitizeText_(data.url); const tag = sanitizeText_(data.tag); const folder = sanitizeText_(data.folder) || '기타'; validateRequired_(title, 'title'); validateRequired_(url, 'url'); validateYoutubeUrl_(url); const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const now = formatNow_(); const id = generateId_(); sheet.appendRow([ id, title, url, tag, folder, now, now ]); return jsonOutput({ ok: true, action: 'add', id, item: { id, title, url, tag, folder, createdAt: now, updatedAt: now } }); } finally { lock.releaseLock(); } } function handleEdit(data) { const sheet = getSheet_(); const id = sanitizeText_(data.id); const title = sanitizeText_(data.title); const url = sanitizeText_(data.url); const tag = sanitizeText_(data.tag); const folder = sanitizeText_(data.folder) || '기타'; validateRequired_(id, 'id'); validateRequired_(title, 'title'); validateRequired_(url, 'url'); validateYoutubeUrl_(url); const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const rowIndex = findRowById_(sheet, id); if (!rowIndex) { throw new Error(`해당 id를 찾을 수 없습니다: ${id}`); } const existingCreatedAt = sheet.getRange(rowIndex, 6).getValue(); const now = formatNow_(); sheet.getRange(rowIndex, 1, 1, 7).setValues([[ id, title, url, tag, folder, existingCreatedAt || now, now ]]); return jsonOutput({ ok: true, action: 'edit', id, item: { id, title, url, tag, folder, createdAt: existingCreatedAt || now, updatedAt: now } }); } finally { lock.releaseLock(); } } function handleDelete(data) { const sheet = getSheet_(); const id = sanitizeText_(data.id); validateRequired_(id, 'id'); const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const rowIndex = findRowById_(sheet, id); if (!rowIndex) { throw new Error(`삭제할 id를 찾을 수 없습니다: ${id}`); } const existing = sheet.getRange(rowIndex, 1, 1, 7).getValues()[0]; sheet.deleteRow(rowIndex); return jsonOutput({ ok: true, action: 'delete', id, deleted: { id: existing[0], title: existing[1], url: existing[2], tag: existing[3], folder: existing[4], createdAt: existing[5], updatedAt: existing[6] } }); } finally { lock.releaseLock(); } } function getSheet_() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { throw new Error(`시트를 찾을 수 없습니다: ${SHEET_NAME}`); } return sheet; } function findRowById_(sheet, id) { const lastRow = sheet.getLastRow(); if (lastRow < 2) return null; const values = sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat(); const index = values.findIndex(v => String(v).trim() === id); if (index === -1) return null; return index + 2; } function generateId_() { const timePart = new Date().getTime(); const randomPart = Math.random().toString(36).slice(2, 8); return `bgm_${timePart}_${randomPart}`; } function formatNow_() { return Utilities.formatDate( new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss' ); } function jsonOutput(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function sanitizeText_(value) { return String(value == null ? '' : value).trim(); } function validateRequired_(value, fieldName) { if (value === '' || value == null) { throw new Error(`${fieldName} 값이 비어 있습니다.`); } } function validateYoutubeUrl_(url) { const patterns = [ /(?:youtube\.com\/watch\?v=)([^&#]+)/i, /(?:youtu\.be\/)([^?&#]+)/i, /(?:youtube\.com\/embed\/)([^?&#]+)/i, /(?:youtube\.com\/shorts\/)([^?&#]+)/i ]; const isValid = patterns.some(regex => regex.test(url)); if (!isValid) { throw new Error('유효한 YouTube URL이 아닙니다.'); } }
